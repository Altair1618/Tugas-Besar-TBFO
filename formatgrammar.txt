Productions:

VAR -> validName;

SUB -> -;
MUL -> *;
DIV -> /;
MOD -> %;

DOUBLE_NEG -> --;
INCDEC -> VAR DOUBLE_PLUS | VAR DOUBLE_NEG;

BOOL -> TRUE | FALSE;
TRUE -> true;
FALSE -> false;
FLOAT -> INT DOT INT;
INT -> validNum;
NUM -> FLOAT | INT;
STRING -> validName;
NULL -> null;
VALUE -> NUM | STRING | BOOL | NULL | VAR;
FIXED_VAL -> NUM | STRING | BOOL | NULL;
DOT -> .;
KURBKI -> (;
KURBKA -> );
KURSKI -> [;
KURSKA -> ];
KURKKI -> {;
